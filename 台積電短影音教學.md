## 系統架構

```
run.py
  │
  ├─[1] data_fetcher.get_all_data()
  │       └─ yfinance 抓取：2330.TW、TSM、TWD=X、NVDA...
  │
  ├─[2] video_maker.make_video(data, output_path)
  │       ├─ make_dashboard_image()   ← matplotlib 畫 1080×1920 儀表板
  │       ├─ effects.py               ← PIL 疊加動態特效（漲跌觸發）
  │       ├─ music_maker.generate_bgm() ← numpy 合成 WAV BGM
  │       └─ ffmpeg                   ← 合成圖 + 音 → MP4
  │
  └─[3] youtube_uploader.upload_video(path, data, type)
          ├─ generate_metadata()      ← 自動產生英文標題/說明/標籤
          ├─ youtube.videos().insert() ← 分塊上傳
          └─ get_or_create_playlist() → add_to_playlist()
```

## BGM 生成 `music_maker.py`

純程式合成，**零版權問題**，根據台積電今日漲跌自動選擇曲風：

| 條件 | 曲風 |
|------|------|
| `change_pct > 0` | Synthwave（活潑、電子合成器）|
| `change_pct <= 0` | Lofi（沉鬱、鋼琴 + 低頻鼓）|

### 呼叫方式
```python
from music_maker import generate_bgm
generate_bgm(duration=30, change_pct=+2.5, output_path="bgm.wav")
```

### 合成原理
```python
SR = 44100   # 取樣率

# 產生正弦波音符
def _sine(freq, dur, amp=0.5):
    t = np.linspace(0, dur, int(SR * dur), endpoint=False)
    return amp * np.sin(2 * np.pi * freq * t)

# 疊加多層音軌：主旋律 + 和聲 + 鼓點 + 殘響
# 最終輸出 int16 WAV
wavfile.write(output_path, SR, (mix * 32767).astype(np.int16))
```

### `data_fetcher.py` — 數據抓取

```python
"""
data_fetcher.py
抓取台積電 (2330) 與全球半導體相關股票的即時數據
"""
import yfinance as yf
import datetime

TICKERS = {
    "TSMC_TW":  "2330.TW",
    "TSMC_ADR": "TSM",
    "FX":       "TWD=X",
    "NVDA": "NVDA", "AAPL": "AAPL", "AMD": "AMD",
    "ASML": "ASML", "AMAT": "AMAT", "INTC": "INTC",
}
SUPPLY_CHAIN = ["NVDA", "AAPL", "AMD", "ASML", "AMAT", "INTC"]


def fetch_price(ticker_symbol: str) -> float:
    try:
        t = yf.Ticker(ticker_symbol)
        hist = t.history(period="2d")
        return float(hist["Close"].iloc[-1]) if not hist.empty else 0.0
    except Exception as e:
        print(f"  [警告] 抓取 {ticker_symbol} 失敗: {e}")
        return 0.0


def fetch_history(ticker_symbol: str, days: int = 5) -> list:
    try:
        t = yf.Ticker(ticker_symbol)
        hist = t.history(period=f"{days + 2}d")
        closes = hist["Close"].tolist()
        return closes[-days:] if len(closes) >= days else closes
    except Exception:
        return []


def fetch_change_pct(ticker_symbol: str) -> float:
    try:
        t = yf.Ticker(ticker_symbol)
        hist = t.history(period="2d")
        if len(hist) < 2:
            return 0.0
        prev = float(hist["Close"].iloc[-2])
        curr = float(hist["Close"].iloc[-1])
        return (curr - prev) / prev * 100
    except Exception:
        return 0.0


def get_all_data() -> dict:
    print("[...] fetching market data...")
    data = {}
    data["tsmc_tw"]  = fetch_price(TICKERS["TSMC_TW"])
    data["tsmc_adr"] = fetch_price(TICKERS["TSMC_ADR"])
    data["fx"]       = fetch_price(TICKERS["FX"])
    if data["fx"] <= 0:
        data["fx"] = 32.0
    adr_in_twd = (data["tsmc_adr"] / 5) * data["fx"]
    data["premium_pct"] = (adr_in_twd - data["tsmc_tw"]) / data["tsmc_tw"] * 100 \
                          if data["tsmc_tw"] > 0 else 0.0
    data["change_pct"]    = fetch_change_pct(TICKERS["TSMC_TW"])
    data["change_tw"]     = data["change_pct"]
    data["change_us"]     = fetch_change_pct(TICKERS["TSMC_ADR"])
    data["tsmc_history"]  = fetch_history(TICKERS["TSMC_TW"], days=5)
    supply = {}
    for name in SUPPLY_CHAIN:
        supply[name] = {"price": fetch_price(TICKERS[name]),
                        "change": fetch_change_pct(TICKERS[name])}
    data["supply_chain"] = supply
    data["timestamp"] = datetime.datetime.now().strftime("%Y-%m-%d %H:%M")
    print(f"  [OK] TW={data['tsmc_tw']:.0f}  ADR={data['tsmc_adr']:.2f}"
          f"  FX={data['fx']:.2f}  Premium={data['premium_pct']:+.2f}%")
    return data
```

```python
video_maker.py
生成 Cyberpunk 風格 YouTube Shorts 短影音 (9:16, 30秒, 1080x1920)
策略：matplotlib 畫高品質儀表板 → ffmpeg 加淡入淡出效果

import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
import matplotlib.gridspec as gridspec
import matplotlib.patheffects as pe
import numpy as np
import os, subprocess, tempfile, shutil
from datetime import datetime

BG="#050510"; GREEN="#00FF41"; RED="#FF0033"; CYAN="#00E5FF"
YELLOW="#FFD600"; WHITE="#FFFFFF"; GRAY="#AABBCC"; DGRAY="#0D0D1A"
W_PX, H_PX, DPI = 1080, 1920, 100
W_IN, H_IN = W_PX/DPI, H_PX/DPI


def _neon_text(ax, x, y, text, color, size, bold=True, **kwargs):
    t = ax.text(x, y, text, color=color, fontsize=size,
                fontweight="bold" if bold else "normal", **kwargs)
    t.set_path_effects([pe.withStroke(linewidth=4, foreground=color, alpha=0.3), pe.Normal()])
    return t

def _color_box(ax, x, y, w, h, edge_color, fill_color=None, lw=1.5, alpha=1.0):
    ax.add_patch(mpatches.FancyBboxPatch((x,y), w, h,
        boxstyle="round,pad=0.02", linewidth=lw,
        edgecolor=edge_color, facecolor=fill_color or DGRAY, alpha=alpha))


def make_dashboard_image(data: dict, output_img: str) -> str:
    tsmc_tw  = data.get("tsmc_tw",  0)
    tsmc_adr = data.get("tsmc_adr", 0)
    fx       = data.get("fx",       32.0)
    premium  = data.get("premium_pct", 0.0)
    history  = data.get("tsmc_history", [tsmc_tw]*5)
    supply   = data.get("supply_chain", {})
    ts       = data.get("timestamp", datetime.now().strftime("%Y-%m-%d %H:%M"))
    history  = np.array(history, dtype=float)
    sc_names   = list(supply.keys()) or ["NVDA","AAPL","AMD","ASML","AMAT","INTC"]
    sc_changes = [supply[n]["change"] for n in sc_names] if supply else [0.0]*6
    sc_prices  = [supply[n].get("price",0) for n in sc_names] if supply else [0.0]*6
    prem_color = GREEN if premium >= 0 else RED
    prem_arrow = "▲" if premium >= 0 else "▼"

    fig = plt.figure(figsize=(W_IN, H_IN), facecolor=BG, dpi=DPI)
    gs  = gridspec.GridSpec(7, 1, figure=fig, hspace=0.06,
                            height_ratios=[1.0,1.8,1.4,0.5,2.2,2.2,0.7],
                            top=0.97, bottom=0.015, left=0.04, right=0.96)
    axes = [fig.add_subplot(gs[i]) for i in range(7)]
    for ax in axes:
        ax.set_facecolor(BG)
        for s in ax.spines.values(): s.set_visible(False)
        ax.set_xticks([]); ax.set_yticks([])
    ax_title, ax_main, ax_prem, ax_sep, ax_heat, ax_chart, ax_foot = axes

    # 標題列
    ax_title.set_xlim(0,1); ax_title.set_ylim(0,1); ax_title.axis("off")
    for y_ in [0.02, 0.98]: ax_title.axhline(y_, color=CYAN, lw=0.5, alpha=0.4)
    _neon_text(ax_title, 0.5, 0.72, "[*] TSMC GLOBAL WAR ROOM",
               CYAN, 32, ha="center", va="center", fontfamily="monospace")
    _neon_text(ax_title, 0.5, 0.28, f">> {ts}  (UTC+8)",
               GRAY, 17, bold=False, ha="center", va="center", fontfamily="monospace")

    # 台股 vs ADR
    ax_main.set_xlim(0,1); ax_main.set_ylim(0,1); ax_main.axis("off")
    _color_box(ax_main, 0.02,0.06,0.44,0.88, GREEN, lw=2)
    ax_main.text(0.24,0.88,"2330.TW", color=GRAY, fontsize=20, ha="center", fontfamily="monospace")
    _neon_text(ax_main, 0.24,0.52, f"NT${tsmc_tw:,.0f}", YELLOW, 46, ha="center", fontfamily="monospace")
    ax_main.text(0.24,0.18,"Taiwan Stock", color=GRAY, fontsize=17, ha="center", fontfamily="monospace")
    _color_box(ax_main, 0.54,0.06,0.44,0.88, CYAN, lw=2)
    ax_main.text(0.76,0.88,"$TSM  ADR", color=GRAY, fontsize=20, ha="center", fontfamily="monospace")
    _neon_text(ax_main, 0.76,0.52, f"${tsmc_adr:.2f}", YELLOW, 46, ha="center", fontfamily="monospace")
    ax_main.text(0.76,0.18,"US Market", color=GRAY, fontsize=17, ha="center", fontfamily="monospace")

    # ADR 溢價
    ax_prem.set_xlim(0,1); ax_prem.set_ylim(0,1); ax_prem.axis("off")
    _color_box(ax_prem, 0.05,0.05,0.90,0.90, prem_color, lw=2.5)
    ax_prem.text(0.5,0.82,"ADR PREMIUM", color=GRAY, fontsize=20, ha="center", fontfamily="monospace")
    _neon_text(ax_prem, 0.5,0.42, f"{prem_arrow}  {abs(premium):.2f} %",
               prem_color, 62, ha="center", fontfamily="monospace")
    ax_prem.text(0.5,0.14, f"FX : {fx:.2f}  (USD/TWD)", color=GRAY, fontsize=16,
                 ha="center", fontfamily="monospace")

    # 供應鏈熱力圖
    ax_sep.set_xlim(0,1); ax_sep.set_ylim(0,1); ax_sep.axis("off")
    ax_sep.text(0.5,0.5,"── SUPPLY CHAIN PULSE ──", color=CYAN, fontsize=19,

    for i,(name,chg,price) in enumerate(zip(sc_names,sc_changes,sc_prices)):
```

## 第六章：背景音樂生成 `music_maker.py`

### 設計思路

音樂不用版權素材，全部用 NumPy 程式合成：

- **上漲（change_pct ≥ 0）** → Synthwave 活潑版（BPM 105，Am-F-C-G 和弦循環）
- **下跌（change_pct < 0）** → Lofi 沉鬱版（BPM 75，Cm-Ab-Eb-Bb 循環 + 乙烯基雜音質感）

音樂結構：Bass（基礎節奏）+ Pad（和弦墊音）+ Arpeggio（琶音旋律）+ 鼓點

### 完整原始碼

```python
"""
music_maker.py
用 numpy 程式生成 30 秒 Synthwave / Lofi 背景音樂
純程式合成，無版權問題
依漲跌自動選擇曲風：上漲 = Synthwave 活潑版，下跌 = Lofi 沉鬱版
"""

import numpy as np
from scipy.io import wavfile
import os

SR = 44100  # 44.1kHz 取樣率

# ── 音符頻率換算 ──────────────────────────────────────────
_NOTE_MAP = {
    'C':0,'C#':1,'Db':1,'D':2,'D#':3,'Eb':3,'E':4,'F':5,
    'F#':6,'Gb':6,'G':7,'G#':8,'Ab':8,'A':9,'A#':10,'Bb':10,'B':11
}

def _freq(note: str) -> float:
    """'A4' → 440.0 Hz"""
    name, octave = note[:-1], int(note[-1])
    semitones = _NOTE_MAP[name] + (octave - 4) * 12
    return 440.0 * (2 ** (semitones / 12))


# ── 波形產生器 ────────────────────────────────────────────

def _sine(freq, dur, amp=0.5):
    t = np.linspace(0, dur, int(SR * dur), endpoint=False)
    return amp * np.sin(2 * np.pi * freq * t)

def _sawtooth(freq, dur, amp=0.3):
    t = np.linspace(0, dur, int(SR * dur), endpoint=False)
    phase = t * freq % 1.0
    return amp * (2 * phase - 1)

def _square(freq, dur, amp=0.2, duty=0.5):
    t = np.linspace(0, dur, int(SR * dur), endpoint=False)
    phase = t * freq % 1.0
    return amp * np.where(phase < duty, 1.0, -1.0)

def _envelope(sig, attack=0.01, decay=0.1, sustain=0.7, release=0.15):
    n = len(sig)
    if n == 0:
        return sig
    env = np.ones(n) * sustain
    a = min(max(1, int(attack * SR)), n)
    d = min(max(1, int(decay * SR)), max(1, n - a))
    r = min(max(1, int(release * SR)), n)
    env[:a]    = np.linspace(0, 1, a)
    env[a:a+d] = np.linspace(1, sustain, d)
    if r < n:
        env[-r:] = np.linspace(sustain, 0, r)
    return sig * env

def _reverb(sig, delay=0.06, decay=0.35):
    """簡單延遲殘響"""
    d = int(delay * SR)
    out = sig.copy()
    if d < len(out):
        out[d:] += sig[:-d] * decay
    return out

def _mix_into(buf, sig, t_sec):
    """把 sig 混入 buf 的 t_sec 時間點"""
    start = int(t_sec * SR)
    if start >= len(buf) or len(sig) == 0:
        return
    end = start + len(sig)
    if end > len(buf):
        sig = sig[:len(buf) - start]
        end = len(buf)
    if len(sig) > 0:
        buf[start:end] += sig


# ── Synthwave 曲風（上漲 / 中性）────────────────────────────

def _gen_synthwave(duration=30, bpm=105) -> np.ndarray:
    total = int(SR * duration)
    buf   = np.zeros(total)
    beat  = 60 / bpm
    bar   = beat * 4

    chords = [
        ('A2', ['A3','C4','E4']),
        ('F2', ['F3','A3','C4']),
        ('C2', ['C3','E4','G4']),
        ('G2', ['G3','B3','D4']),
    ]
    arp_seqs = [
        ['A4','C5','E5','C5'],
        ['F4','A4','C5','A4'],
        ['C4','E5','G5','E5'],
        ['G4','B4','D5','B4'],
    ]

    t = 0.0; idx = 0
    while t < duration:
        bass_note, pad_notes = chords[idx % len(chords)]
        arp_notes = arp_seqs[idx % len(arp_seqs)]
        dur = min(bar, duration - t)

        sig = _sine(_freq(bass_note), dur, amp=0.30)
        sig = _envelope(sig, 0.02, 0.08, 0.55, 0.18)
        _mix_into(buf, sig, t)

        for n in pad_notes:
            sig = _sine(_freq(n), dur, amp=0.07)
            sig = _envelope(sig, 0.12, 0.15, 0.60, 0.25)
            _mix_into(buf, sig, t)

        step = beat / 4
        for i, n in enumerate(arp_notes * int(dur / step / len(arp_notes) + 1)):
            at = t + i * step
            if at >= duration: break
            note_dur = min(step * 0.9, duration - at)
            sig = _sawtooth(_freq(n), note_dur, amp=0.14)
            sig = _envelope(sig, 0.003, 0.06, 0.35, 0.08)
            _mix_into(buf, sig, at)

        for b in range(int(dur / beat) + 1):
            bt = t + b * beat
            if bt >= duration: break
            kick_dur = min(0.18, duration - bt)
            kt = np.linspace(0, kick_dur, int(SR * kick_dur), endpoint=False)
            kick = 0.40 * np.sin(2 * np.pi * (80 - 60 * kt / kick_dur) * kt)
            kick *= np.exp(-kt * 25)
            _mix_into(buf, kick, bt)

        for h in range(int(dur / (beat / 2)) + 1):
            ht = t + h * beat / 2
            if ht >= duration: break
            hh_dur = min(0.04, duration - ht)
            noise = 0.08 * np.random.default_rng(h + idx * 100).standard_normal(int(SR * hh_dur))
            noise *= np.exp(-np.linspace(0, hh_dur * 40, len(noise)))
            _mix_into(buf, noise, ht)

        t += bar; idx += 1

    buf = _reverb(buf, delay=0.07, decay=0.30)
    return buf


# ── Lofi 曲風（下跌 / 警示）────────────────────────────────

def _gen_lofi(duration=30, bpm=75) -> np.ndarray:
    total = int(SR * duration)
    buf   = np.zeros(total)
    beat  = 60 / bpm
    bar   = beat * 4

    chords = [
        ('C2', ['C3','Eb4','G4']),
        ('Ab2',['Ab3','C4','Eb4']),
        ('Eb2',['Eb3','G4','Bb4']),
        ('Bb2',['Bb3','D4','F4']),
    ]
    arp_seqs = [
        ['C4','G4','Eb4','G4'],
        ['Ab4','Eb4','C4','Eb4'],
        ['Eb4','Bb4','G4','Bb4'],
        ['Bb4','F4','D4','F4'],
    ]

    t = 0.0; idx = 0
    rng = np.random.default_rng(7)

    while t < duration:
        bass_note, pad_notes = chords[idx % len(chords)]
        arp_notes = arp_seqs[idx % len(arp_seqs)]
        dur = min(bar, duration - t)

        sig = _sine(_freq(bass_note), dur, amp=0.28)
        sig = _envelope(sig, 0.05, 0.15, 0.50, 0.25)
        _mix_into(buf, sig, t)

        for n in pad_notes:
            sig = _sine(_freq(n), dur, amp=0.055)
            sig = _envelope(sig, 0.20, 0.20, 0.55, 0.30)
            _mix_into(buf, sig, t)

        step = beat / 2
        for i, n in enumerate(arp_notes * int(dur / step / len(arp_notes) + 1)):
            at = t + i * step
            if at >= duration: break
            note_dur = min(step * 0.85, duration - at)
            sig = _sine(_freq(n), note_dur, amp=0.10)
            sig = _envelope(sig, 0.02, 0.10, 0.45, 0.20)
            _mix_into(buf, sig, at)

        for b in range(int(dur / beat) + 1):
            bt = t + b * beat
            if bt >= duration: break
            kick_dur = min(0.22, duration - bt)
            kt = np.linspace(0, kick_dur, int(SR * kick_dur), endpoint=False)
            kick = 0.25 * np.sin(2 * np.pi * (65 - 40 * kt / kick_dur) * kt)
            kick *= np.exp(-kt * 18)
            _mix_into(buf, kick, bt)

        t += bar; idx += 1

    buf = _reverb(buf, delay=0.10, decay=0.40)
    vinyl = rng.standard_normal(total) * 0.015
    buf += vinyl
    return buf


# ── 主入口 ────────────────────────────────────────────────

def generate_bgm(
    duration: int     = 30,
    change_pct: float = 0.0,
    output_path: str  = "output/bgm.wav"
) -> str:
    """
    根據漲跌幅自動選擇曲風並生成背景音樂
    change_pct > 0  → Synthwave（上漲）
    change_pct < 0  → Lofi（下跌）
    回傳 WAV 路徑
    """
    os.makedirs(os.path.dirname(output_path) or ".", exist_ok=True)

    if change_pct >= 0:
        style = "Synthwave"
        buf = _gen_synthwave(duration, bpm=105)
    else:
        style = "Lofi"
        buf = _gen_lofi(duration, bpm=75)

    # Normalize → 16-bit PCM
    mx = np.max(np.abs(buf))
    if mx > 0:
        buf = buf / mx * 0.82
    audio = (buf * 32767).astype(np.int16)

    wavfile.write(output_path, SR, audio)
    print(f"  [OK] BGM generated [{style}]: {output_path}")
    return output_path


if __name__ == "__main__":
    generate_bgm(30, change_pct=+4.0, output_path="output/test_synthwave.wav")
    generate_bgm(30, change_pct=-4.0, output_path="output/test_lofi.wav")
    print("試聽：output/test_synthwave.wav  /  output/test_lofi.wav")
```

## 第七章：YouTube 上傳模組 `youtube_uploader.py`

### 功能概覽

- OAuth2 授權（token 快取到 `youtube_token.pickle`，不用每次重新登入）
- 根據當日數據自動生成英文標題、說明、標籤
- **分塊上傳（Resumable Upload）**：5 MB 一塊，斷網可續傳
- 自動建立或取得播放清單（ID 快取到 `playlist_ids.json`）

### OAuth2 授權流程

```
第一次執行時：
  程式開啟瀏覽器 → 登入 Google → 同意授權
  → token 儲存到 youtube_token.pickle
  
之後執行：
  直接讀取 pickle，自動 refresh（有效期通常 1 小時，refresh_token 永久有效）
```

```python
            }
        }
    ).execute()
    print(f"   已加入播放清單")


def upload_video(video_path: str, data: dict, video_type: str = "warroom") -> str:
    """上傳影片到 YouTube，並自動加入對應播放清單，回傳影片 URL"""
    if not os.path.exists(video_path):
        print(f"找不到影片: {video_path}")
        return ""

    meta = generate_metadata(data, video_type)
    print(f"   上傳中... 標題: {meta['title']}")

    youtube = get_youtube_service()

    body = {
        "snippet": {
            "title":           meta["title"],
            "description":     meta["description"],
            "tags":            meta["tags"],
            "categoryId":      CATEGORY_SCIENCE,
            "defaultLanguage": "en",
        },
        "status": {
            "privacyStatus":            "public",
            "selfDeclaredMadeForKids":  False,
            "madeForKids":              False,
        }
    }

    media = MediaFileUpload(video_path, mimetype="video/mp4",
                            resumable=True, chunksize=1024 * 1024 * 5)
    request = youtube.videos().insert(
        part="snippet,status", body=body, media_body=media)

    response = None
    while response is None:
        status, response = request.next_chunk()
        if status:
            print(f"   上傳進度: {int(status.progress()*100)}%", end="\r")

    video_id  = response["id"]
    video_url = f"https://www.youtube.com/shorts/{video_id}"
    print(f"\n   上傳成功！URL: {video_url}")

    try:
        playlist_id = get_or_create_playlist(youtube, video_type)
        add_to_playlist(youtube, video_id, playlist_id)
    except Exception as e:
        print(f"   加入播放清單失敗（影片已上傳成功）: {e}")

    return video_url
```

## 第九章：疑難排解

### 常見問題

**Q: `yfinance` 抓不到台股數據？**
台股收盤後（13:35）到隔天開盤前，`2330.TW` 回傳的是昨日收盤價，這是正常現象。建議在台股收盤後 15:30 執行以取得最終收盤數據。

**Q: ffmpeg 找不到？**
確認已安裝並加入 PATH：
```bash
# Windows
winget install ffmpeg

# 驗證
ffmpeg -version
```

**Q: YouTube 上傳時出現 `quotaExceeded`？**
YouTube Data API v3 預設每日配額為 10,000 單位，一次上傳約消耗 1,600 單位。每天最多可上傳約 6 支影片。若需更多，可在 Google Cloud Console 申請配額提升。

**Q: `client_secret.json` 放哪裡？**
放在 `run.py` 同一個目錄下。第一次執行 `python run.py` 時，程式會自動開啟瀏覽器要求授權，授權後 token 存入 `youtube_token.pickle`，之後不需再手動操作。

**Q: 影片上傳成功但沒有出現在 Shorts？**
確認影片符合 Shorts 格式：時長 ≤ 60 秒、解析度為 9:16（1080×1920）。本系統輸出的影片為 30 秒 1080×1920，應自動被 YouTube 歸類為 Shorts。
